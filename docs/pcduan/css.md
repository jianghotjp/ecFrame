@support-ie-version: 8;     // 支持的浏览器版本
@support-html5: true;       // 支持HTML5

@default-border-width: 1px;     // 默认边框宽度

@code: Monaco, Menlo, Consolas, "Courier New", "FontAwesome", monospace;
